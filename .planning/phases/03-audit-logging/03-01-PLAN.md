       - CREATE TABLE IF NOT EXISTS audit_logs with columns: id TEXT PK, timestamp TEXT NOT NULL,
         user_id TEXT NOT NULL, action_type TEXT NOT NULL, resource_type TEXT, resource_id TEXT,
         device_id TEXT NOT NULL, success INTEGER NOT NULL DEFAULT 1, error_message TEXT,
         previous_hash TEXT NOT NULL, entry_hash TEXT NOT NULL UNIQUE,

       - CREATE INDEX IF NOT EXISTS idx_audit_logs_timestamp ON audit_logs(timestamp)
       - CREATE INDEX IF NOT EXISTS idx_audit_logs_user_id ON audit_logs(user_id)

       - CREATE INDEX IF NOT EXISTS idx_audit_logs_action_type ON audit_logs(action_type)
       - BEFORE UPDATE trigger: audit_logs_prevent_update using RAISE(ABORT, 'Audit log entries are immutable')
       - BEFORE DELETE trigger: audit_logs_prevent_delete using RAISE(ABORT, 'Audit log entries cannot be deleted')
       Note: rusqlite_migration does not support M::up with multiple semicolon-separated statements
       in all versions. If that causes issues, concatenate as a single string (the existing project
       does this — see Migration 2 which has CREATE TABLE + CREATE INDEX in one M::up string).

    4. src-tauri/src/audit/entry.rs:
       - use sha2::{Sha256, Digest}; use hex; use uuid; use chrono;
       - ActionType enum with variants: Create, Read, List, Update, Delete, Login, Logout,
         LockSession, UnlockSession, BreakGlassActivate, BreakGlassDeactivate, MfaSetup, MfaDisable
       - impl ActionType { pub fn as_str(&self) -> &'static str { ... } }
       - AuditEntry struct (pub fields matching schema)
       - fn compute_entry_hash(entry: &AuditEntry) -> String
         Canonical: "{}|{}|{}|{}|{}|{}|{}|{}|{}|{}" with fields in fixed order
         (previous_hash, id, timestamp, user_id, action_type.as_str(),
          resource_type.unwrap_or(""), resource_id.unwrap_or(""),
          device_id, success as "1"/"0", error_message.unwrap_or(""))
       - fn get_last_hash(conn: &rusqlite::Connection) -> Result<String, AppError>
         SELECT entry_hash FROM audit_logs ORDER BY rowid DESC LIMIT 1;
         QueryReturnedNoRows => Ok("GENESIS".to_string())
       - pub fn write_audit_entry(conn: &rusqlite::Connection, user_id: &str,
           action_type: ActionType, resource_type: Option<&str>, resource_id: Option<&str>,
           device_id: &str, success: bool, error_message: Option<&str>) -> Result<(), AppError>
         Steps: gen id, gen timestamp, get_last_hash, build AuditEntry, compute_entry_hash, INSERT

       - pub fn read_audit_entries(conn: &rusqlite::Connection, user_id_filter: Option<&str>,
           limit: i64, offset: i64) -> Result<Vec<AuditLogRow>, AppError>

         ORDER BY timestamp DESC LIMIT ? OFFSET ?
       - pub fn verify_audit_chain(conn: &rusqlite::Connection) -> Result<bool, AppError>
         SELECT all rows ORDER BY rowid ASC. For each row, recompute entry_hash from its
         fields and compare to stored value. Also verify row[n].previous_hash == row[n-1].entry_hash.
         Return Ok(true) if chain intact, Ok(false) if any mismatch found.
